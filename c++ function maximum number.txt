return_type function_name(arg_type_1 arg_1, arg_type_2 arg_2, ...) {
    ...
    ...
    ...
    [if return_type is non void]
        return something of type `return_type`;
}

int sum_of_four(int a, int b, int c, int d) {

a += b is equivalent to a = a + b;
